# u-body

# 说明

# 通用视图组件，所有页面都需要调用

# 无 scroll-view

<u-body />

# 有 scroll-view

# wxml

<u-body
  scroll
  bg=" {{ bg }} "
  btn="{{ btn }}"
  load="{{ Boolean }}"
  reHeight="{{ reHeight }}"
  class="{{ ios ? 'ios' : 'android' }}"
  end="{{ end }}"
  more="{{ more }}"
  bind:onRefresh="onRefresh"
  bind:loadmore="loadMore"
></u-body>
